231 /**
232  * \brief Get status of a Group of Pins (atomic)
233  *
234  * Reads the level of the given pins. Bit 0 of the returned data corresponds to
235  * GPIO0 level, bit 1 to GPIO1 level. and so on. Bits corresponding to masked
236  * pins (corresponding bit of gpios parameter set to zero) are returned as 0.
237  *
238  * This is an atomic operation.
239  *
240  * @param[in] gpioport GPIO block register address base @ref gpio_reg_base
241  * @param[in] gpios @ref gpio_pin_id. Any combination of pins may be specified
242  * by OR'ing then together.
243  *
244  * @return The level of the GPIO port. The pins not specified in gpios are
245  * masked to zero.
246  */
247 static inline u8 gpio_read(u32 gpioport, u8 gpios)
248 {
249  return GPIO_DATA(gpioport)[gpios];
250 }
251 
252 /**
253  * \brief Set level of a Group of Pins (atomic)
254  *
255  * Sets the level of the given pins. Bit 0 of the data parameter corresponds to
256  * GPIO0, bit 1 to GPIO1. and so on. Maskedpins (corresponding bit of gpios
257  * parameter set to zero) are returned not affected.
258  *
259  * This is an atomic operation.
260  *
261  * @param[in] gpioport GPIO block register address base @ref gpio_reg_base
262  * @param[in] gpios @ref gpio_pin_id. Any combination of pins may be specified
263  * by OR'ing then together.
264  * @param[in] data Level to set pin to. Bit 0 of data corresponds to GPIO0, bit
265  * 1 to GPIO1. and so on.
266  */
267 static inline void gpio_write(u32 gpioport, u8 gpios, u8 data)
268 {
269  /* ipaddr[9:2] mask the bits to be set, hence the array index */
270  GPIO_DATA(gpioport)[gpios] = data;
271 }

354 /**
355  * \brief Mark interrupt as serviced
356  *

358  * cleared, then execution will jump back to the start of the ISR after the ISR
359  * returns.
360  *
361  * @param[in] gpioport GPIO block register address base @ref gpio_reg_base
