163 void rcc_wait_for_sysclk_status(osc_t osc)
164 {
165  switch (osc) {
166  case PLL:
167  while ((RCC_CFGR & ((1 << 1) | (1 << 0))) != RCC_CFGR_SWS_PLL);
168  break;
169  case HSE:
170  while ((RCC_CFGR & ((1 << 1) | (1 << 0))) != RCC_CFGR_SWS_HSE);
171  break;
172  case HSI:
173  while ((RCC_CFGR & ((1 << 1) | (1 << 0))) != RCC_CFGR_SWS_HSI);
174  break;
175  default:
176  /* Shouldn't be reached. */
177  break;
178  }
179 }

364 void rcc_clock_setup_hse_3v3(const clock_scale_t *clock)
365 {
366  /* Enable internal high-speed oscillator. */
367  rcc_osc_on(HSI);
368  rcc_wait_for_osc_ready(HSI);
369 
370  /* Select HSI as SYSCLK source. */
371  rcc_set_sysclk_source(RCC_CFGR_SW_HSI);
372 
373  /* Enable external high-speed oscillator 8MHz. */
374  rcc_osc_on(HSE);
375  rcc_wait_for_osc_ready(HSE);
376 
377  /*
378  * Set prescalers for AHB, ADC, ABP1, ABP2.
379  * Do this before touching the PLL (TODO: why?).
380  */
381  rcc_set_hpre(clock->hpre);
382  rcc_set_ppre1(clock->ppre1);
383  rcc_set_ppre2(clock->ppre2);
384 
385  rcc_set_main_pll_hse(clock->pllm, clock->plln,
386  clock->pllp, clock->pllq);
387 
388  /* Enable PLL oscillator and wait for it to stabilize. */
389  rcc_osc_on(PLL);
390  rcc_wait_for_osc_ready(PLL);
391 
392  /* Configure flash settings. */
393  flash_set_ws(clock->flash_config);
394 
395  /* Select PLL as SYSCLK source. */
396  rcc_set_sysclk_source(RCC_CFGR_SW_PLL);
397 
398  /* Wait for PLL clock to be selected. */
399  rcc_wait_for_sysclk_status(PLL);
400 
401  /* Set the peripheral clock frequencies used. */
402  rcc_ppre1_frequency = clock->apb1_frequency;
403  rcc_ppre2_frequency = clock->apb2_frequency;
404 }
