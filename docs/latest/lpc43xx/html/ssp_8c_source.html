57 /*
58 * SSP Init function
59 */
60 void ssp_init(ssp_num_t ssp_num,
61  ssp_datasize_t data_size,
62  ssp_frame_format_t frame_format,
63  ssp_cpol_cpha_t cpol_cpha_format,
64  u8 serial_clock_rate,
65  u8 clk_prescale,
66  ssp_mode_t mode,
67  ssp_master_slave_t master_slave,
68  ssp_slave_option_t slave_option)
69 {
70  u32 ssp_port;
71  u32 clock;
72 
73  if(ssp_num == SSP0_NUM)
74  {
75  ssp_port = SSP0;
76  }else
77  {
78  ssp_port = SSP1;
79  }
80 
81  /* use PLL1 as clock source for SSP1 */
82  CGU_BASE_SSP1_CLK = (CGU_SRC_PLL1<<CGU_BASE_CLK_SEL_SHIFT) | (1<<CGU_AUTOBLOCK_CLOCK_BIT);
83 
84  /* Disable SSP before to configure it */
85  SSP_CR1(ssp_port) = 0x0;
86 
87  /* Configure SSP */
88  clock = serial_clock_rate;
89  SSP_CPSR(ssp_port) = clk_prescale;
90  SSP_CR0(ssp_port) = (data_size | frame_format | cpol_cpha_format | (clock<<8) );
91 
92  /* Enable SSP */
93  SSP_CR1(ssp_port) = (SSP_ENABLE | mode | master_slave | slave_option);
94 }

97 * This Function Wait until Data RX Ready, and return Data Read from SSP.

99 u16 ssp_read(ssp_num_t ssp_num)
100 {
101  u32 ssp_port;
102 
103  if(ssp_num == SSP0_NUM)
104  {
105  ssp_port = SSP0;
106  }else
107  {
108  ssp_port = SSP1;
109  }
110  /* Wait Until Data Received (Rx FIFO not Empty) */
111  while( (SSP_SR(ssp_port) & SSP_SR_RNE) == 0);
112 
113  return SSP_DR(ssp_port);
114 }

131 /* This Function Wait Data TX Ready, and Write Data to SSP */
132 void ssp_write(ssp_num_t ssp_num, u16 data)
133 {
134  u32 ssp_port;
135 
136  if(ssp_num == SSP0_NUM)
137  {
138  ssp_port = SSP0;
139  }else
140  {
141  ssp_port = SSP1;
142  }
143 
144  /* Wait Until FIFO not full */
145  while( (SSP_SR(ssp_port) & SSP_SR_TNF) == 0);
146 
147  SSP_DR(ssp_port) = data;
148 
149  /* Wait for not busy, since we're controlling CS# of
150  * devices manually and need to wait for the data to
151  * be sent. It may also be important to wait here
152  * in case we're configuring devices via SPI and also
153  * with GPIO control -- we need to know when SPI
154  * commands are effective before altering a device's
155  * state with GPIO. I'm thinking the MAX2837, for
156  * example...
157  */
158  ssp_wait_until_not_busy(ssp_num);
159 }
