43 /*
44 * UART Init function
45 */
46 void uart_init(uart_num_t uart_num,
47  uart_databit_t data_nb_bits,
48  uart_stopbit_t data_nb_stop,
49  uart_parity_t data_parity,
50  uint16_t uart_divisor,
51  uint8_t uart_divaddval,
52  uint8_t uart_mulval)
53 {
54  uint32_t lcr_config;
55  uint32_t uart_port;
56 
57  uart_port = uart_num;
58 
59  switch(uart_num)
60  {
61  case UART0_NUM:
62  /* use PLL1 as clock source for UART0 */
63  CGU_BASE_UART0_CLK = (CGU_SRC_PLL1<<UART_CGU_BASE_CLK_SEL_SHIFT) | (1<<UART_CGU_AUTOBLOCK_CLOCK_BIT);
64  break;
65 
66  case UART1_NUM:
67  /* use PLL1 as clock source for UART1 */
68  CGU_BASE_UART1_CLK = (CGU_SRC_PLL1<<UART_CGU_BASE_CLK_SEL_SHIFT) | (1<<UART_CGU_AUTOBLOCK_CLOCK_BIT);
69  break;
70 
71  case UART2_NUM:
72  /* use PLL1 as clock source for UART2 */
73  CGU_BASE_UART2_CLK = (CGU_SRC_PLL1<<UART_CGU_BASE_CLK_SEL_SHIFT) | (1<<UART_CGU_AUTOBLOCK_CLOCK_BIT);
74  break;
75 
76  case UART3_NUM:
77  /* use PLL1 as clock source for UART3 */
78  CGU_BASE_UART3_CLK = (CGU_SRC_PLL1<<UART_CGU_BASE_CLK_SEL_SHIFT) | (1<<UART_CGU_AUTOBLOCK_CLOCK_BIT);
79  break;
80 
81  default:
82  return; /* error */
83  }
84 
85  /* FIFOs RX/TX Enabled and Reset RX/TX FIFO (DMA Mode is also cleared) */
86  UART_FCR(uart_port) = ( UART_FCR_FIFO_EN | UART_FCR_RX_RS | UART_FCR_TX_RS);
87  /* Disable FIFO */
88  UART_FCR(uart_port) = 0;
89 
90  // Dummy read (to clear existing data)
91  while (UART_LSR(uart_port) & UART_LSR_RDR ) {
92  dummy_read = UART_RBR(uart_port);
93  }
94 
95  /* Wait end of TX & disable TX */
96  UART_TER(uart_port) = UART_TER_TXEN;
97 
98  /* Wait for current transmit complete */
99  while (!(UART_LSR(uart_port) & UART_LSR_THRE));
100 
101  /* Disable Tx */
102  UART_TER(uart_port) = 0;
103 
104  /* Disable interrupt */
105  UART_IER(uart_port) = 0;
106 
107  /* Set LCR to default state */
108  UART_LCR(uart_port) = 0;
109 
110  /* Set ACR to default state */
111  UART_ACR(uart_port) = 0;
112 
113  /* Dummy Read to Clear Status */
114  dummy_read = UART_LSR(uart_port);
115 
116  /*
117  Table 835. USART Fractional Divider Register:
118  UARTbaudrate = PCLK / ( 16* (((256*DLM)+ DLL)*(1+(DivAddVal/MulVal))) )
119  The value of MULVAL and DIVADDVAL should comply to the following conditions:
120  1. 1 <= MULVAL <= 15
121  2. 0 <= DIVADDVAL <= 14
122  3. DIVADDVAL < MULVAL
123  */
124 
125  /* Set DLAB Bit */
126  UART_LCR(uart_port) |= UART_LCR_DLAB_EN;
127  UART_DLM(uart_port) = UART_LOAD_DLM(uart_divisor);
128  UART_DLL(uart_port) = UART_LOAD_DLL(uart_divisor);
129  /* Clear DLAB Bit */
130  UART_LCR(uart_port) &= (~UART_LCR_DLAB_EN) & UART_LCR_BITMASK;
131  UART_FDR(uart_port) = (UART_FDR_MULVAL(uart_mulval) | UART_FDR_DIVADDVAL(uart_divaddval)) & UART_FDR_BITMASK;
132 
133  /* Read LCR config & Force Enable of Divisor Latches Access */
134  lcr_config = (UART_LCR(uart_port) & UART_LCR_DLAB_EN) & UART_LCR_BITMASK;
135  lcr_config |= data_nb_bits; /* Set Nb Data Bits */
136  lcr_config |= data_nb_stop; /* Set Nb Stop Bits */
137  lcr_config |= data_parity; /* Set Data Parity */
138 
139  /* Write LCR (only 8bits) */
140  UART_LCR(uart_port) = (lcr_config & UART_LCR_BITMASK);
141 
142  /* Enable TX */
143  UART_TER(uart_port) = UART_TER_TXEN;
144 }

146 /*
147 * This Function return if data are received or not received.
148 */
149 uart_rx_data_ready_t uart_rx_data_ready(uart_num_t uart_num)
150 {
151  uint32_t uart_port;
152  uint8_t uart_status;
153  uart_rx_data_ready_t data_ready;
154 
155  uart_port = uart_num;
156 
157  uart_status = UART_LSR(uart_port) & 0xFF;
158 
159  /* Check Error */
160  if ((uart_status & UART_LSR_ERROR_MASK) == 0) {
161  /* No errors check if data is ready */
162  if ((uart_status & UART_LSR_RDR) == 0) {
163  data_ready = UART_RX_NO_DATA;
164  } else {
165  data_ready = UART_RX_DATA_READY;
166  }
167  } else {
168  /* UART Error */
169  data_ready = UART_RX_DATA_ERROR;
170  }
171 
172  return data_ready;
173 }

175 /*
176 * This Function Wait until Data RX Ready, and return Data Read from UART.
177 */
178 uint8_t uart_read(uart_num_t uart_num)
179 {
180  uint32_t uart_port;
181  uint8_t uart_val;
182 
183  uart_port = uart_num;
184 
185  /* Wait Until Data Received (Rx Data Not Ready) */
186  while ((UART_LSR(uart_port) & UART_LSR_RDR) == 0);
187 
188  uart_val = (UART_RBR(uart_port) & UART_RBR_MASKBIT);
189 
190  return uart_val;
191 }

196 uint8_t uart_read_timeout(uart_num_t uart_num, uint32_t rx_timeout_nb_cycles, uart_error_t* error)
197 {
198  uint32_t uart_port;
199  uint8_t uart_val;
200  uint32_t counter;
201 
202  uart_port = uart_num;
203 
204  /* Wait Until Data Received (Rx Data Not Ready) */
205  counter = 0;
206  while ((UART_LSR(uart_port) & UART_LSR_RDR) == 0)
207  {
208  if (rx_timeout_nb_cycles>0) {
209  counter++;
210  if (counter>=rx_timeout_nb_cycles) {
211  *error = UART_TIMEOUT_ERROR;
212  return 0;
213  }
214  }
215  }
216 
217  uart_val = (UART_RBR(uart_port) & UART_RBR_MASKBIT);
218 
219  /* Clear error */
220  *error = UART_NO_ERROR;
221 
222  return uart_val;
223 }

225 /* This Function Wait Data TX Ready, and Write Data to UART
226  if rx_timeout_nb_cycles = 0 Infinite wait
227 */
228 void uart_write(uart_num_t uart_num, uint8_t data)
229 {
230  uint32_t uart_port;
231 
232  uart_port = uart_num;
233 
234  /* Wait Until FIFO not full */
235  while ((UART_LSR(uart_port) & UART_LSR_THRE) == 0);
236 
237  UART_THR(uart_port) = data;
