44 void i2c0_init(void)
45 {
46  /* enable input on SCL and SDA pins */
47  SCU_SFSI2C0 = SCU_I2C0_NOMINAL;
48 
49  /* use IRC as clock source for APB1 (including I2C0) */
50  CGU_BASE_APB1_CLK = (CGU_SRC_IRC << CGU_BASE_CLK_SEL_SHIFT);
51 
52  /* FIXME assuming we're on IRC at 12 MHz */
53 
54  /* 400 kHz I2C */
55  I2C0_SCLH = 15;
56  I2C0_SCLL = 15;
57 
58  /* 100 kHz I2C */
59  /*
60  I2C0_SCLH = 60;
61  I2C0_SCLL = 60;

63 
64  /* clear the control bits */
65  I2C0_CONCLR = (I2C_CONCLR_AAC | I2C_CONCLR_SIC
66  | I2C_CONCLR_STAC | I2C_CONCLR_I2ENC);

68  /* enable I2C0 */

72 /* transmit start bit */
73 void i2c0_tx_start(void)
74 {
75  I2C0_CONCLR = I2C_CONCLR_SIC;
76  I2C0_CONSET = I2C_CONSET_STA;
77  while (!(I2C0_CONSET & I2C_CONSET_SI));
78  I2C0_CONCLR = I2C_CONCLR_STAC;
79 }

81 /* transmit data byte */
82 void i2c0_tx_byte(u8 byte)
83 {
84  if (I2C0_CONSET & I2C_CONSET_STA)
85  I2C0_CONCLR = I2C_CONCLR_STAC;
86  I2C0_DAT = byte;
87  I2C0_CONCLR = I2C_CONCLR_SIC;
88  while (!(I2C0_CONSET & I2C_CONSET_SI));
89 }

91 /* receive data byte */
92 u8 i2c0_rx_byte(void)
93 {
94  if (I2C0_CONSET & I2C_CONSET_STA)
95  I2C0_CONCLR = I2C_CONCLR_STAC;
96  I2C0_CONCLR = I2C_CONCLR_SIC;
97  while (!(I2C0_CONSET & I2C_CONSET_SI));
