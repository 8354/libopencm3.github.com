59 /** @brief IWDG Set Period in Milliseconds
60 
61 The countdown period is converted into count and prescale values. The maximum
62 period is 32.76 seconds; values above this are truncated. Periods less than 1ms
63 are not supported by this library.
64 
65 A delay of up to 5 clock cycles of the LSI clock (about 156 microseconds)
66 can occasionally occur if the prescale or preload registers are currently busy
67 loading a previous value.
68 
69 @param[in] period u32 Period in milliseconds (< 32760) from a watchdog reset until
70 a system reset is issued.
71 */
72 
73 void iwdg_set_period_ms(u32 period)
74 {
75 u32 count, prescale, reload, exponent;
76 /* Set the count to represent ticks of the 32kHz LSI clock */
77  count = (period << 5);
78 /* Strip off the first 12 bits to get the prescale value required */
79  prescale = (count >> 12);
80  if (prescale > 256) {exponent = IWDG_PR_DIV256; reload = COUNT_MASK;}
81  else if (prescale > 128) {exponent = IWDG_PR_DIV256; reload = (count >> 8);}
82  else if (prescale > 64) {exponent = IWDG_PR_DIV128; reload = (count >> 7);}
83  else if (prescale > 32) {exponent = IWDG_PR_DIV64; reload = (count >> 6);}
84  else if (prescale > 16) {exponent = IWDG_PR_DIV32; reload = (count >> 5);}
85  else if (prescale > 8) {exponent = IWDG_PR_DIV16; reload = (count >> 4);}
86  else if (prescale > 4) {exponent = IWDG_PR_DIV8; reload = (count >> 3);}
87  else {exponent = IWDG_PR_DIV4; reload = (count >> 2);}
88 /* Avoid the undefined situation of a zero count */
89  if (count == 0) count = 1;
90 
91  while (iwdg_prescaler_busy());
92  IWDG_KR = IWDG_KR_UNLOCK;
93  IWDG_PR = exponent;
94  while (iwdg_reload_busy());
95  IWDG_KR = IWDG_KR_UNLOCK;
96  IWDG_RLR = (reload & COUNT_MASK);
97 }
