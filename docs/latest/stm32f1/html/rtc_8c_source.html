41 void rtc_awake_from_off(enum rcc_osc clock_source)
42 {
43  uint32_t reg32;
44 
45  /* Enable power and backup interface clocks. */
46  RCC_APB1ENR |= (RCC_APB1ENR_PWREN | RCC_APB1ENR_BKPEN);
47 
48  /* Enable access to the backup registers and the RTC. */
49  PWR_CR |= PWR_CR_DBP;
50 
51  /*
52  * Reset the backup domain, clears everything RTC related.
53  * If not wanted use the rtc_awake_from_standby() function.
54  */
55  rcc_backupdomain_reset();
56 
57  switch (clock_source) {
58  case LSE:
59  /* Turn the LSE on and wait while it stabilises. */
60  RCC_BDCR |= RCC_BDCR_LSEON;
61  while ((reg32 = (RCC_BDCR & RCC_BDCR_LSERDY)) == 0);
62 
63  /* Choose LSE as the RTC clock source. */
64  RCC_BDCR &= ~((1 << 8) | (1 << 9));
65  RCC_BDCR |= (1 << 8);
66  break;
67  case LSI:
68  /* Turn the LSI on and wait while it stabilises. */
69  RCC_CSR |= RCC_CSR_LSION;
70  while ((reg32 = (RCC_CSR & RCC_CSR_LSIRDY)) == 0);
71 
72  /* Choose LSI as the RTC clock source. */
73  RCC_BDCR &= ~((1 << 8) | (1 << 9));
74  RCC_BDCR |= (1 << 9);
75  break;
76  case HSE:
77  /* Turn the HSE on and wait while it stabilises. */
78  RCC_CR |= RCC_CR_HSEON;
79  while ((reg32 = (RCC_CR & RCC_CR_HSERDY)) == 0);
80 
81  /* Choose HSE as the RTC clock source. */
82  RCC_BDCR &= ~((1 << 8) | (1 << 9));
83  RCC_BDCR |= (1 << 9) | (1 << 8);
84  break;
85  case PLL:
86  case PLL2:
87  case PLL3:
88  case HSI:
89  /* Unusable clock source, here to prevent warnings. */
90  /* Turn off clock sources to RTC. */
91  RCC_BDCR &= ~((1 << 8) | (1 << 9));
92  break;
93  }
94 
95  /* Enable the RTC. */
96  RCC_BDCR |= RCC_BDCR_RTCEN;
97 
98  /* Wait for the RSF bit in RTC_CRL to be set by hardware. */
99  RTC_CRL &= ~RTC_CRL_RSF;
100  while ((reg32 = (RTC_CRL & RTC_CRL_RSF)) == 0);
101 
102  /* Wait for the last write operation to finish. */
103  /* TODO: Necessary? */
104  while ((reg32 = (RTC_CRL & RTC_CRL_RTOFF)) == 0);
105 }

107 void rtc_enter_config_mode(void)
108 {
109  uint32_t reg32;
110 
111  /* Wait until the RTOFF bit is 1 (no RTC register writes ongoing). */
112  while ((reg32 = (RTC_CRL & RTC_CRL_RTOFF)) == 0);
113 
114  /* Enter configuration mode. */
115  RTC_CRL |= RTC_CRL_CNF;
116 }
