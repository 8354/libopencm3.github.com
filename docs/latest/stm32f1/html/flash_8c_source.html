116 void flash_program_word(uint32_t address, uint32_t data)
117 {
118  /* Ensure that all flash operations are complete. */
119  flash_wait_for_last_operation();
120 
121  /* Enable writes to flash. */
122  FLASH_CR |= FLASH_CR_PG;
123 
124  /* Program the first half of the word. */
125  MMIO16(address) = (uint16_t)data;
126 
127  /* Wait for the write to complete. */
128  flash_wait_for_last_operation();
129 
130  /* Program the second half of the word. */
131  MMIO16(address + 2) = data >> 16;
132 
133  /* Wait for the write to complete. */
134  flash_wait_for_last_operation();
135 
136  /* Disable writes to flash. */
137  FLASH_CR &= ~FLASH_CR_PG;
138 }

176 void flash_erase_option_bytes(void)
177 {
178  flash_wait_for_last_operation();
179 
180  if ((FLASH_CR & FLASH_CR_OPTWRE) == 0) {
181  flash_unlock_option_bytes();
182  }
183 
184  FLASH_CR |= FLASH_CR_OPTER; /* Enable option byte erase. */
