71 /** @brief SPI Reset.
72 
73 The SPI peripheral and all its associated configuration registers are placed in the
74 reset condition. The reset is effected via the RCC peripheral reset system.
75 
76 @param[in] spi_peripheral Unsigned int32. SPI peripheral identifier @ref spi_reg_base.
77 */
78 
79 void spi_reset(u32 spi_peripheral)
80 {
81  switch (spi_peripheral) {
82  case SPI1:
83  rcc_peripheral_reset(&RCC_APB2RSTR, RCC_APB2RSTR_SPI1RST);
84  rcc_peripheral_clear_reset(&RCC_APB2RSTR, RCC_APB2RSTR_SPI1RST);
85  break;
86  case SPI2:
87  rcc_peripheral_reset(&RCC_APB1RSTR, RCC_APB1RSTR_SPI2RST);
88  rcc_peripheral_clear_reset(&RCC_APB1RSTR, RCC_APB1RSTR_SPI2RST);
89  break;
90 #if defined(STM32F1) || defined(STM32F2) || defined(STM32F3) || defined(STM32F4)
91  case SPI3:
92  rcc_peripheral_reset(&RCC_APB1RSTR, RCC_APB1RSTR_SPI3RST);
93  rcc_peripheral_clear_reset(&RCC_APB1RSTR, RCC_APB1RSTR_SPI3RST);
94  break;
95 #endif
96  }
97 }

100 /** @brief Configure the SPI as Master.
101 
102 The SPI peripheral is configured as a master with communication parameters
103 baudrate, data format 8/16 bits, frame format lsb/msb first, clock polarity
104 and phase. The SPI enable, CRC enable and CRC next controls are not affected.
105 These must be controlled separately.
106 
107 @todo NSS pin handling.
108 
109 @param[in] spi Unsigned int32. SPI peripheral identifier @ref spi_reg_base.
110 @param[in] br Unsigned int32. Baudrate @ref spi_baudrate.
111 @param[in] cpol Unsigned int32. Clock polarity @ref spi_cpol.
112 @param[in] cpha Unsigned int32. Clock Phase @ref spi_cpha.
113 @param[in] dff Unsigned int32. Data frame format 8/16 bits @ref spi_dff.
114 @param[in] lsbfirst Unsigned int32. Frame format lsb/msb first @ref spi_lsbfirst.
115 @returns int. Error code.
116 */
117 
118 int spi_init_master(u32 spi, u32 br, u32 cpol, u32 cpha, u32 dff, u32 lsbfirst)
119 {
120  u32 reg32 = SPI_CR1(spi);
121 
122  /* Reset all bits omitting SPE, CRCEN and CRCNEXT bits. */
123  reg32 &= SPI_CR1_SPE | SPI_CR1_CRCEN | SPI_CR1_CRCNEXT;
124 
125  reg32 |= SPI_CR1_MSTR; /* Configure SPI as master. */
126 
127  reg32 |= br; /* Set baud rate bits. */
128  reg32 |= cpol; /* Set CPOL value. */
129  reg32 |= cpha; /* Set CPHA value. */
130  reg32 |= dff; /* Set data format (8 or 16 bits). */
131  reg32 |= lsbfirst; /* Set frame format (LSB- or MSB-first). */
132 
133  /* TODO: NSS pin handling. */
134 
135  SPI_CR1(spi) = reg32;
136 
137  return 0; /* TODO */
138 }

174 /*-----------------------------------------------------------------------------*/
175 /** @brief SPI Clean Disable.
176 
177 Disable the SPI peripheral according to the procedure in section 23.3.8 of the
178 reference manual. This prevents corruption of any ongoing transfers and
179 prevents the BSY flag from becoming unreliable.
180 
181 @param[in] spi Unsigned int32. SPI peripheral identifier @ref spi_reg_base.
182 @returns data Unsigned int16. 8 or 16 bit data from final read.
183 */
184 
185 u16 spi_clean_disable(u32 spi)
186 {
187  /* Wait to receive last data */
188  while (!(SPI_SR(spi) & SPI_SR_RXNE))
189  ;
190 
191  u16 data = SPI_DR(spi);
192 
193  /* Wait to transmit last data */
194  while (!(SPI_SR(spi) & SPI_SR_TXE))
195  ;
196 
197  /* Wait until not busy */
198  while (SPI_SR(spi) & SPI_SR_BSY)
199  ;
200 
201  SPI_CR1(spi) &= ~SPI_CR1_SPE;
202 
203  return data;
204 }
