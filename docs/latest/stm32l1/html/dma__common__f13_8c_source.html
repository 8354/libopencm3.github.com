1 /** @addtogroup dma_file
2 
3 @author @htmlonly &copy; @endhtmlonly 2010 Thomas Otto <tommi@viadmin.org>
4 
5 This library supports the DMA Control System in the STM32 series of ARM Cortex
6 Microcontrollers by ST Microelectronics.
7 
8 Up to two DMA controllers are supported. 12 DMA channels are allocated 7 to
9 the first DMA controller and 5 to the second. Each channel is connected to
10 between 3 and 6 hardware peripheral DMA signals in a logical OR arrangement.
11 
12 DMA transfers can be configured to occur between peripheral and memory in
13 any combination including memory to memory. Circular mode transfers are
14 also supported in transfers involving a peripheral. An arbiter is provided
15 to resolve priority DMA requests. Transfers can be made with 8, 16 or 32 bit
16 words.

44 /** @brief DMA Channel Reset
45 
46 The channel is disabled and configuration registers are cleared.
47 
48 @param[in] dma unsigned int32. DMA controller base address: DMA1 or DMA2
49 @param[in] channel unsigned int8. Channel number: 1-7 for DMA1 or 1-5 for DMA2
50 */
51 
52 void dma_channel_reset(u32 dma, u8 channel)
53 {
54  /* Disable channel and reset config bits. */
55  DMA_CCR(dma, channel) = 0;
56  /* Reset data transfer number. */
57  DMA_CNDTR(dma, channel) = 0;
58  /* Reset peripheral address. */
59  DMA_CPAR(dma, channel) = 0;
60  /* Reset memory address. */
61  DMA_CMAR(dma, channel) = 0;
62  /* Reset interrupt flags. */
63  DMA_IFCR(dma) |= DMA_IFCR_CIF(channel);
64 }

67 /** @brief DMA Channel Clear Interrupt Flag
68 
69 The interrupt flag for the channel is cleared. More than one interrupt for the
70 same channel may be cleared by using the logical OR of the interrupt flags.
71 
72 @param[in] dma unsigned int32. DMA controller base address: DMA1 or DMA2
73 @param[in] channel unsigned int8. Channel number: @ref dma_ch
74 @param[in] interrupts unsigned int32. Logical OR of interrupt numbers: @ref dma_if_offset
75 */
76 
77 void dma_clear_interrupt_flags(u32 dma, u8 channel, u32 interrupts)
78 {
79 /* Get offset to interrupt flag location in channel field */
80  u32 flags = (interrupts << DMA_FLAG_OFFSET(channel));
81  DMA_IFCR(dma) = flags;
82 }

87 The interrupt flag for the channel is returned.

89 @param[in] dma unsigned int32. DMA controller base address: DMA1 or DMA2

94

96 {
97 /* get offset to interrupt flag location in channel field. */
98  u32 flag = (interrupt << DMA_FLAG_OFFSET(channel));
99  return ((DMA_ISR(dma) & flag) > 0);
100 }
