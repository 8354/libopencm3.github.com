52 /** Initialize the USB device controller hardware of the STM32. */
53 static usbd_device *stm32f107_usbd_init(void)
54 {
55  OTG_FS_GINTSTS = OTG_FS_GINTSTS_MMIS;
56 
57  OTG_FS_GUSBCFG |= OTG_FS_GUSBCFG_PHYSEL;
58  /* Enable VBUS sensing in device mode and power down the PHY. */
59  OTG_FS_GCCFG |= OTG_FS_GCCFG_VBUSBSEN | OTG_FS_GCCFG_PWRDWN;
60 
61  /* Wait for AHB idle. */
62  while (!(OTG_FS_GRSTCTL & OTG_FS_GRSTCTL_AHBIDL)) ;
63  /* Do core soft reset. */
64  OTG_FS_GRSTCTL |= OTG_FS_GRSTCTL_CSRST;
65  while (OTG_FS_GRSTCTL & OTG_FS_GRSTCTL_CSRST) ;
66 
67  /* Force peripheral only mode. */
68  OTG_FS_GUSBCFG |= OTG_FS_GUSBCFG_FDMOD | OTG_FS_GUSBCFG_TRDT_MASK;
69 
70  /* Full speed device. */
71  OTG_FS_DCFG |= OTG_FS_DCFG_DSPD;
72 
73  /* Restart the PHY clock. */
74  OTG_FS_PCGCCTL = 0;
75 
76  OTG_FS_GRXFSIZ = stm32f107_usb_driver.rx_fifo_size;
77  usbd_dev.fifo_mem_top = stm32f107_usb_driver.rx_fifo_size;
78 
79  /* Unmask interrupts for TX and RX. */
80  OTG_FS_GAHBCFG |= OTG_FS_GAHBCFG_GINT;
81  OTG_FS_GINTMSK = OTG_FS_GINTMSK_ENUMDNEM |
82  OTG_FS_GINTMSK_RXFLVLM |
83  OTG_FS_GINTMSK_IEPINT |
84  OTG_FS_GINTMSK_USBSUSPM |
85  OTG_FS_GINTMSK_WUIM |
86  OTG_FS_GINTMSK_SOFM;
87  OTG_FS_DAINTMSK = 0xF;
88  OTG_FS_DIEPMSK = OTG_FS_DIEPMSK_XFRCM;
89 
90  return &usbd_dev;
91 }
