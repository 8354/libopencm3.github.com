10 Each I/O port has 16 individually configurable bits. Many I/O pins share GPIO
11 functionality with a number of alternate functions and must be configured to the
12 alternate function mode if these are to be accessed. A feature is available to
13 remap alternative functions to a limited set of alternative pins in the event
14 of a clash of requirements.
15 
16 The data registers associated with each port for input and output are 32 bit with
17 the upper 16 bits unused. The output buffer must be written as a 32 bit word, but
18 individual bits may be set or reset separately in atomic operations to avoid race
19 conditions during interrupts. Bits may also be individually locked to prevent
20 accidental configuration changes. Once locked the configuration cannot be changed
21 until after the next reset.
22 
23 Each port bit can be configured as analog or digital input, the latter can be
24 floating or pulled up or down. As outputs they can be configured as either
25 push-pull or open drain, digital I/O or alternate function, and with maximum
26 output speeds of 2MHz, 10MHz, or 50MHz.
27 
28 On reset all ports are configured as digital floating input.

80 /** @brief Set GPIO Pin Mode
81 
82 Sets the Pin Direction and Analog/Digital Mode, and Output Pin Pullup,
83 for a set of GPIO pins on a given GPIO port.
84 
85 @param[in] gpioport Unsigned int32. Port identifier @ref gpio_port_id
86 @param[in] mode Unsigned int8. Pin mode @ref gpio_mode
87 @param[in] pull_up_down Unsigned int8. Pin pullup/pulldown configuration @ref gpio_pup
88 @param[in] gpios Unsigned int16. Pin identifiers @ref gpio_pin_id
89  If multiple pins are to be set, use bitwise OR '|' to separate them.
90 */
91 void gpio_mode_setup(u32 gpioport, u8 mode, u8 pull_up_down, u16 gpios)
92 {
93  u16 i;
94  u32 moder, pupd;
95 
96  /*
97  * We want to set the config only for the pins mentioned in gpios,
98  * but keeping the others, so read out the actual config first.
99  */
100  moder = GPIO_MODER(gpioport);
101  pupd = GPIO_PUPDR(gpioport);
102 
103  for (i = 0; i < 16; i++) {
104  if (!((1 << i) & gpios))
105  continue;
106 
107  moder &= ~GPIO_MODE_MASK(i);
108  moder |= GPIO_MODE(i, mode);
109  pupd &= ~GPIO_PUPD_MASK(i);
110  pupd |= GPIO_PUPD(i, pull_up_down);
111  }
112 
113  /* Set mode and pull up/down control registers. */
114  GPIO_MODER(gpioport) = moder;
115  GPIO_PUPDR(gpioport) = pupd;
116 }

153 /** @brief Set GPIO Alternate Function Selection
154 
155 Set the alternate function mapping number for each pin. Most pins have alternate
156 functions associated with them. When set to AF mode, a pin may be used for one of
157 its allocated alternate functions selected by the number given here. To determine
158 the number to be used for the desired function refer to the individual datasheet
159 for the particular device. A table is given under the Pin Selection chapter.
160 
161 Note that a number of pins may be set but only with a single AF number. In practice
162 this would rarely be useful as each pin is likely to require a different number.
163 
164 @param[in] gpioport Unsigned int32. Port identifier @ref gpio_port_id
165 @param[in] alt_func_num Unsigned int8. Pin alternate function number @ref gpio_af_num
166 @param[in] gpios Unsigned int16. Pin identifiers @ref gpio_pin_id
167  If multiple pins are to be set, use bitwise OR '|' to separate them.
168 */
169 void gpio_set_af(u32 gpioport, u8 alt_func_num, u16 gpios)
170 {
171  u16 i;
172  u32 afrl, afrh;
173 
174  afrl = GPIO_AFRL(gpioport);
175  afrh = GPIO_AFRH(gpioport);
176 
177  for (i = 0; i < 8; i++) {
178  if (!((1 << i) & gpios))
179  continue;
180  afrl &= ~GPIO_AFR_MASK(i);
181  afrl |= GPIO_AFR(i, alt_func_num);
182  }
183 
184  for (i = 8; i < 16; i++) {
185  if (!((1 << i) & gpios))
186  continue;
187  afrh &= ~GPIO_AFR_MASK(i - 8);
188  afrh |= GPIO_AFR(i - 8, alt_func_num);
189  }
190 
191  GPIO_AFRL(gpioport) = afrl;
